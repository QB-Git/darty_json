Json.fromList(List<dynamic> list) {
  _rawValue = <dynamic>[];

  try {
    int i = 0;
    for (var item in list) {
      _set(i++, item);
    }
  } on JsonException catch (error) {
    exception = exception ?? error;
  }
}
